static WFDropdownItem? selectedItemById(
    List<WFDropdownItem> list, String id) {
  WFDropdownItem? selectedItem;
  for (WFDropdownItem obj in list) {
    if (Utils.equals(id, obj.id)) {
      selectedItem = obj;
      break;
    }
  }
  if (selectedItem == null && list.isNotEmpty) {
    selectedItem = list[0];
  }
  return selectedItem;
}
